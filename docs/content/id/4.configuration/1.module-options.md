Configure Nuxt PDF Kit globally in your `nuxt.config.ts`.

## Basic Configuration

```ts [nuxt.config.ts]
export default defineNuxtConfig({
  modules: ['nuxt-pdf-kit'],
  pdfKit: {
    toolbar: {
      // toolbar options
    },
  },
})
```

## Default Values

All toolbar options are enabled by default except `openFile`:

```ts [nuxt.config.ts]
export default defineNuxtConfig({
  modules: ['nuxt-pdf-kit'],
  pdfKit: {
    toolbar: {
      sidebar: true,
      pageNavigation: true,
      zoom: true,
      search: true,
      rotate: true,
      openFile: false,    // Disabled by default
      print: true,
      download: true,
      fullscreen: true,
      themeToggle: true,
      moreOptions: true,
    },
  },
})
```
