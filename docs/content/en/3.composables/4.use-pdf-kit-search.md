Provides full-text search functionality with match highlighting.

## Usage

```ts
const {
  searchQuery,
  searchMatches,
  currentMatchIndex,
  isSearching,
  caseSensitive,
  wholeWords,
  search,
  nextMatch,
  prevMatch,
  clearSearch,
  toggleCaseSensitive,
  toggleWholeWords,
} = usePdfKitSearch()
```

## Return Values

| Property | Type | Description |
|----------|------|-------------|
| `searchQuery` | `Ref<string>` | Current search query |
| `searchMatches` | `Ref<SearchMatch[]>` | Array of search matches |
| `currentMatchIndex` | `Ref<number>` | Index of current match |
| `isSearching` | `Ref<boolean>` | Search in progress |
| `caseSensitive` | `Ref<boolean>` | Case sensitive search |
| `wholeWords` | `Ref<boolean>` | Match whole words only |
| `search` | `(query: string) => Promise<void>` | Perform search |
| `nextMatch` | `() => void` | Go to next match |
| `prevMatch` | `() => void` | Go to previous match |
| `clearSearch` | `() => void` | Clear search results |
| `toggleCaseSensitive` | `() => void` | Toggle case sensitivity |
| `toggleWholeWords` | `() => void` | Toggle whole words |

## SearchMatch Interface

```ts
interface SearchMatch {
  pageIndex: number
  matchIndex: number
  text: string
}
```

## Example

```vue
<template>
  <div class="flex items-center gap-2">
    <UInput 
      v-model="query" 
      placeholder="Search..." 
      @keyup.enter="performSearch"
    />
    <span v-if="searchMatches.length">
      {{ currentMatchIndex + 1 }} / {{ searchMatches.length }}
    </span>
    <UButton @click="prevMatch">Prev</UButton>
    <UButton @click="nextMatch">Next</UButton>
    <UButton @click="clearSearch">Clear</UButton>
  </div>
</template>

<script setup>
const query = ref('')
const {
  searchMatches,
  currentMatchIndex,
  search,
  nextMatch,
  prevMatch,
  clearSearch,
} = usePdfKitSearch()

async function performSearch() {
  await search(query.value)
}
</script>
```
